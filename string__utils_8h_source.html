41 template<UINT CodePage>
42 wstring _to_wstring(const string& data)
43 {
44  size_t nLengthW = data.length() + 1;
45  wchar_t* str = new wchar_t[nLengthW];
46  memset(str, 0, nLengthW * sizeof(wchar_t));
47  MultiByteToWideChar(CodePage, 0, data.c_str(), -1, str, (int)nLengthW);
48  wstring res(str);
49  delete[] str;
50  return res;
51 }

53 template<UINT CodePage>
54 string wstring_to_(const wstring& data)
55 {
56  size_t size = data.length();
57  if (size)
58  {
59  char* str = new char[2 * size + 1]; // for russian utf8 string case
60  memset(str, 0, 2 * size + 1);
61  WideCharToMultiByte(CodePage, 0, data.c_str(), -1, str, 2 * (int)size + 1, NULL, NULL);
62  string res(str);
63  delete[] str;
64  return res;
65  }
66  else
67  return string();
68 }
