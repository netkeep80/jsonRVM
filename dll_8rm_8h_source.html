83 class DLLs : public map<string, DLL>

85 public:
86 DLLs() = default;
87 ~DLLs()
88 {
89 for (auto& dll : *this)

93 FreeLibrary(dll.second.handle);

95 dlclose(dll.second.handle);

98 dll.second.Init = nullptr;

102 void LoadDict(const string &LibName)
103 {
104 DLLs &it = *this;
105 if (find(LibName) == end())
106 it[LibName] = DLL();
107
108 if (!it[LibName].handle)
109 {
110 // перезагружаем либу
111#ifdef _WIN32
112 it[LibName].handle = LoadLibrary(utf8_to_wstring(LibName).c_str());
113#else
114 it[LibName].handle = dlopen(LibName.c_str(), RTLD_LAZY);
115#endif
116
117 if (it[LibName].handle)
118 {
119#ifdef _WIN32
120 (FARPROC &)it[LibName].Init = GetProcAddress(it[LibName].handle, IMPORT_RELATIONS_MODEL);
121#else
122 it[LibName].Init = reinterpret_cast<InitDict>(dlsym(it[LibName].handle, IMPORT_RELATIONS_MODEL));
123#endif
124 if (!it[LibName].Init)
125 throw json({{__func__, LibName + " does't has function "s + IMPORT_RELATIONS_MODEL}});
126 }
127 else
128 {
129 it[LibName].Init = nullptr;
130#ifdef _WIN32
131 throw json({{__func__, "can't load '" + LibName + "' dictionary"s}});
132#else
133 const char* err = dlerror();
134 throw json({{__func__, "can't load '" + LibName + "' dictionary: "s + (err ? err : "unknown error")}});
135#endif
136 }
137 }
138 }
139 } LoadedDLLs;

170 void jsonLoadDLL(vm &rmvm, vm_ctx &$)
171 {
172 try
173 {
174 if ($.obj.is_object())
175 {
176 if ($.obj.count("PathFolder") && $.obj.count("FileName"))
177 {
178 string FullFileName = $.obj["PathFolder"].get<string>() + $.obj["FileName"].get<string>();
179 $.sub = LoadAndInitDict(FullFileName, rmvm);
180 return;
181 }
182 }
183 }
184 catch (json &j)
185 {
186 $.throw_json(__func__, j);
187 }
188 catch (json::exception &e)
189 {
190 $.throw_json(__func__, "json::exception: "s + e.what() + ", id: "s + to_string(e.id));
191 }
192 catch (std::exception &e)
193 {
194 $.throw_json(__func__, "std::exception: "s + e.what());
195 }
196 catch (...)
197 {
198 $.throw_json(__func__, "unknown exception"s);
199 }
200
201 $.rel = false;
202 $.throw_json(__func__, "$obj must be json object with PathFolder, FileName properties!"s);
203 }
