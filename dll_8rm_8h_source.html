74 class DLLs : public map<string, DLL>
75 {
76 public:
77 DLLs() = default;
78 ~DLLs()
79 {
80 for each (auto dll in *this)
81 if (dll.second.handle)
82 {
83 FreeLibrary(dll.second.handle);
84 dll.second.handle = nullptr;
85 dll.second.Init = nullptr;
86 }
87 }

89 void LoadDict(const string &LibName)
90 {
91 DLLs &it = *this;
92 if (find(LibName) == end())
93 it[LibName] = DLL();
94
95 if (!it[LibName].handle)
96 {
97 // перезагружаем либу
98 it[LibName].handle = LoadLibrary(utf8_to_wstring(LibName).c_str());
99
100 if (it[LibName].handle)
101 {
102 (FARPROC &)it[LibName].Init = GetProcAddress(it[LibName].handle, IMPORT_RELATIONS_MODEL);
103 if (!it[LibName].Init)
104 throw json({{__func__, LibName + " does't has function "s + IMPORT_RELATIONS_MODEL}});
105 }
106 else
107 {
108 it[LibName].Init = nullptr;
109 throw json({{__func__, "can't load '" + LibName + "' dictionary"s}});
110 }
111 }
112 }
113 } LoadedDLLs;

115 const string &LoadAndInitDict(const string &LibName, vm &Ent)
116 {
117 LoadedDLLs.LoadDict(LibName);
118 return LoadedDLLs[LibName].Init(Ent);
119 }

144 void jsonLoadDLL(vm &rmvm, vm_ctx &$)
145 {
146 try
147 {
148 if ($.obj.is_object())
149 {
150 if ($.obj.count("PathFolder") && $.obj.count("FileName"))
151 {
152 string FullFileName = $.obj["PathFolder"].get<string>() + $.obj["FileName"].get<string>();
153 $.sub = LoadAndInitDict(FullFileName, rmvm);
154 return;
155 }
156 }
157 }
158 catch (json &j)
159 {
160 $.throw_json(__func__, j);
161 }
162 catch (json::exception &e)
163 {
164 $.throw_json(__func__, "json::exception: "s + e.what() + ", id: "s + to_string(e.id));
165 }
166 catch (std::exception &e)
167 {
168 $.throw_json(__func__, "std::exception: "s + e.what());
169 }
170 catch (...)
171 {
172 $.throw_json(__func__, "unknown exception"s);
173 }
174
175 $.rel = false;
176 $.throw_json(__func__, "$obj must be json object with PathFolder, FileName properties!"s);
177 }
