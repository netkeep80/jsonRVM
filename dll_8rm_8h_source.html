64  void LoadDict(const string& LibName)
65  {
66  DLLs& it = *this;
67  if (find(LibName) == end()) it[LibName] = DLL();
68 
69  if (!it[LibName].handle)
70  {
71  // перезагружаем либу
72  it[LibName].handle = LoadLibrary(utf8_to_wstring(LibName).c_str());
73 
74  if (it[LibName].handle)
75  {
76  (FARPROC&)it[LibName].Init = GetProcAddress(it[LibName].handle, IMPORT_RELATIONS_MODEL);
77  if (!it[LibName].Init)
78  throw json({ { __FUNCTION__, LibName + " does't has function "s + IMPORT_RELATIONS_MODEL} });
79  }
80  else
81  {
82  it[LibName].Init = nullptr;
83  throw json({ { __FUNCTION__, "can't load '" + LibName + "' dictionary"s} });
84  }
85  }
86  }

98  void jsonLoadDLL(vm& rmvm, vm_ctx& $)
99  {
100  try

110  }
111  }
112  catch (json& j) { $.throw_json(__FUNCTION__, j); }
113  catch (json::exception& e) { $.throw_json(__FUNCTION__, "json::exception: "s + e.what() + ", id: "s + to_string(e.id)); }
114  catch (std::exception& e) { $.throw_json(__FUNCTION__, "std::exception: "s + e.what()); }
115  catch (...) { $.throw_json(__FUNCTION__, "unknown exception"s); }
116 
117  $.rel = false;
118  $.throw_json(__FUNCTION__, "$obj must be json object with PathFolder, FileName properties!"s);
119  }
