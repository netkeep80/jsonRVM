53  inline json xor_array(const json& a, const json& b)
54  {
55  json ar = json::array();
56  bool not_is_null = false;
57 
58  for (size_t i = 0; (i < a.size()) && (i < b.size()); i++)
59  {
60  ar[i] = a.at(i) ^ b.at(i);
61  not_is_null = not_is_null || !ar[i].is_null();
62  }
63 
64  if (a.size() > b.size())
65  {
66  for (size_t i = b.size(); i < a.size(); i++)
67  {
68  ar[i] = a.at(i);
69  not_is_null = not_is_null || !ar[i].is_null();
70  }
71  }
72  else if (a.size() < b.size())
73  {
74  for (size_t i = a.size(); i < b.size(); i++)
75  {
76  ar[i] = b.at(i);
77  not_is_null = not_is_null || !ar[i].is_null();
78  }
79  }
80 
81  if (not_is_null) return ar;
82  else return json();
83  }

95  inline json xor_object(const json& a, const json& b)
96  {
97  json obj = json(), field;
98 
99  for (auto& it = a.begin(); it != a.end(); ++it)
100  {
101  const string& key = it.key();
102  const json& val = it.value();
103  if (b.count(key))
104  {
105  field = val ^ b.at(key);
106  if (!field.is_null()) obj[key] = field;
107  }
108  else obj[key] = val;
109  }
110 
111  for (auto& it = b.begin(); it != b.end(); ++it)
112  {
113  const string& key = it.key();
114  const json& val = it.value();
115  if (a.count(key))
116  {
117  field = a.at(key) ^ val;
118  if (!field.is_null()) obj[key] = field;
119  }
120  else obj[key] = val;
121  }
122 
123  return obj;
124  }

163  {
