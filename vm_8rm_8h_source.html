99  template<class val_type>
100  static void ref_in_json_to(json*& jptr, const string& it)
101  {
102  json& ref = *jptr;
103 
104  switch (ref.type())
105  {
106  case json::value_t::object:
107  jptr = &ref[it];
108  return;
109 
110  case json::value_t::array:
111  jptr = &ref[std::stoul(it)];
112  return;
113 
114  case json::value_t::null:
115  if constexpr (val_type::is_lval)
116  {
117  int& _Errno_ref = errno; // Nonzero cost, pay it once
118  const char* _Ptr = it.c_str();
119  char* _Eptr;
120  _Errno_ref = 0;
121  unsigned index = strtoul(_Ptr, &_Eptr, 10);
122 
123  if (_Errno_ref == ERANGE)
124  throw json({ {__FUNCTION__, it} });
125 
126  if (_Ptr == _Eptr)
127  jptr = &ref[it];
128  else
129  jptr = &ref[index];
130 
131  return;
132  }
133 
134  default:
135  throw json({ {__FUNCTION__, it} });
136  }
137  }

139  template<class val_type>
140  json& string_ref_to(vm_ctx& $, const string& str)
141  {
142  json* jptr;
143  size_t len = str.length();
144  size_t pos = str.find_first_of('/', 0);
145 
146  if (pos == str.npos) pos = len;
147 
148  size_t prev = pos + 1;
149  string it = str.substr(0, pos);
150  //auto it = string_view(str.c_str(), pos);
151 
152  SWITCH(it)
153  {
154  CASE("$$$$ent") : jptr = &$.$.$.$.ent; break;
155  CASE("$$$$sub") : jptr = &$.$.$.$.sub; break;
156  CASE("$$$$obj") : jptr = &$.$.$.$.obj; break;
157  CASE("$$$$rel") : jptr = &$.$.$.$.rel; break;
158  CASE("$$$ent") : jptr = &$.$.$.ent; break;
159  CASE("$$$sub") : jptr = &$.$.$.sub; break;
160  CASE("$$$obj") : jptr = &$.$.$.obj; break;
161  CASE("$$$rel") : jptr = &$.$.$.rel; break;
162  CASE("$$ent") : jptr = &$.$.ent; break;
163  CASE("$$sub") : jptr = &$.$.sub; break;
164  CASE("$$obj") : jptr = &$.$.obj; break;
165  CASE("$$rel") : jptr = &$.$.rel; break;
166  CASE("$ent") : jptr = &$.ent; break;
167  CASE("$sub") : jptr = &$.sub; break;
168  CASE("$obj") : jptr = &$.obj; break;
169  CASE("$rel") : jptr = &$.rel; break;
170 
171  DEFAULT:
172  json& ref = *this;
173  assert(ref.is_object());
174  auto res = ref.find(it);
175 
176  if (res == ref.end())
177  {
178  try { this->get_entity(ref[it], it); }
179  catch (json& j) { throw json({ {__FUNCTION__, j} }); }
180  catch (invalid_argument& e) { throw json({ {__FUNCTION__, "property '"s + str + "' invalid_argument, " + e.what()} }); }
181  catch (out_of_range& e) { throw json({ {__FUNCTION__, "property '"s + str + "' out_of_range, " + e.what()} }); }
182  catch (...) { throw json({ {__FUNCTION__, "property '"s + str + "' does not exist!"} }); }
183 
184  res = ref.find(it);
185  if (res == ref.end())
186  throw json({ {__FUNCTION__, "entity '"s + it + "' does not exist in relations model!"s} });
187  }
188  jptr = &res.value();
189  }
190 
191  while (prev < len)
192  {
193  pos = str.find_first_of('/', prev);
194  if (pos == string::npos) pos = len;
195  string it = str.substr(prev, pos - prev);
196  prev = pos + 1;
197  try { ref_in_json_to<val_type>(jptr, it); }
198  catch (json& j) { throw json({ {__FUNCTION__, j} }); }
199  catch (invalid_argument& e) { throw json({ {__FUNCTION__, "property '"s + str + "' invalid_argument, " + e.what()} }); }
200  catch (out_of_range& e) { throw json({ {__FUNCTION__, "property '"s + str + "' out_of_range, " + e.what()} }); }
201  catch (...) { throw json({ {__FUNCTION__, "property '"s + str + "' does not exist!"} }); }
202  }
203 
204  return *jptr;
205  }

253  // Исполнение сущности либо json байткода
254  // имеет прототип отличный от других контроллеров и не является контроллером
255  // рекурсивно раскручивает структуру проекции контроллера доходя до простых json или вызовов скомпилированных сущностей
256  void exec_ent(vm_ctx& $, json& ent)
257  {
258  binary_view_map_t& dict = *this;
259  auto it = dict.find(&ent);
260 
261  if (it != dict.end()) // это скомпилированная сущность?
262  {
263  try { (*it->second)(*this, $); }
264  catch (json& j) { throw j; }
265  catch (json::exception& e) { throw json("json::exception: "s + e.what() + ", id: "s + to_string(e.id)); }
266  catch (std::exception& e) { throw json("std::exception: "s + e.what()); }
267  catch (...) { throw json("unknown exception"s); }
268  return;
269  }
270  else switch (ent.type())
271  {
272  case json::value_t::string: // иерархический путь к json значению
273  try { exec_ent($, string_ref_to<rval>($, ent.get_ref<string const&>())); }
274  catch (json& j) { throw json({ {ent.get_ref<string const&>(), j} }); }
275  return;
276 
277  case json::value_t::array: // лямбда вектор, который управляет последовательным изменением проекции сущности
278  for (auto it = ent.begin(); it != ent.end(); ++it)
279  try { exec_ent($, *it); }
280  catch (json & j) { throw json({ {"["s + to_string(it - ent.begin()) + "]"s, j} }); }
281  return;
282 
283  case json::value_t::object:
284  {
285  auto& rel = ent.find("$rel");
286  auto& end = ent.end();
287 
288  if (rel != end) // это сущность, которую надо исполнить в новом контексте?
289  {
290  auto& obj = ent.find("$obj");
291  auto& sub = ent.find("$sub");
292 
293  try {
294  exec_ent(
295  vm_ctx(
296  $.rel,
297  obj == end ? $.rel : val_or_ref_to<rval>($, *obj),
298  sub == end ? $.rel : val_or_ref_to<lval>($, *sub),
299  ent,
300  $),
301  val_or_ref_to<rval>($, *rel)
302  );
303  }
304  catch (json & j) { throw json({ {"$rel"s, j} }); }
305  }
306  else // контроллер это лямбда структура, которая управляет параллельным проецированием сущностей
307  {
308  for (auto& it : ent.items()) try
309  {
310  exec_ent(
311  vm_ctx(
312  string_ref_to<lval>($, it.key()),
313  $.obj,
314  $.sub,
315  $.ent,
316  $.$),
317  val_or_ref_to<rval>($, it.value())
318  );
319  }
320  catch (json & j) { throw json({ {it.key(), j} }); }
321  }
322  return;
323  }
324 
325  default: // остальные простые типы есть результат исполнения отношения
326  $.rel = ent;
327 
328  case json::value_t::null: // null - означает отсутствие отношения, т.е. неизменность проекции
329  return;
330  }
331  }
